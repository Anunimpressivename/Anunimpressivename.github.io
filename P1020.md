#include <bits/stdc++.h>
using namespace std;

class Analyzer{
private:
    vector<int> increase;  // 树状数组，用于维护以每个值结尾的最长严格递增子序列长度
    vector<int> decrease;  // 树状数组，用于维护以每个值开头的最长非严格递减子序列长度（通过值域反转实现）

    // 查询树状数组中 [1, idx] 范围内的最大值
    // pattern == true 表示查询 increase 数组，否则查询 decrease 数组
    int getMax(int idx, bool pattern){
        int ans = 0;
        vector<int>& tree = pattern? increase: decrease;
        for(int i = idx; i > 0; i -= (i & -i)) ans = max(ans, tree[i]);
        return ans;
    }

    // 在树状数组的 idx 位置更新值
    // pattern == true 表示更新 increase（递增子序列），否则更新 decrease（递减子序列）
    void add(int idx, bool pattern){
        // 对于递增：查询 [1, idx-1] 的最大 LIS 长度；对于递减：查询 [1, idx] 的最大 LDS 长度（因值已反转）
        int q = pattern? (idx > 1? getMax(idx - 1, true): 0): getMax(idx, false);
        int val = q + 1;  // 当前位置可形成的子序列长度
        vector<int>& tree = pattern? increase: decrease;
        // 树状数组单点更新：将 val 向上传播，维护区间最大值
        for(int i = idx; i <= (int)tree.size() - 1; i += (i & -i)){
            tree[i] = max(tree[i], val);
        }
    }

public:
    // 构造函数：对输入数组 org 预处理，构建 increase 和 decrease 树状数组
    Analyzer(const vector<int>& org): increase(org.size() + 1, 0), decrease(org.size() + 1, 0){
        int n = org.size();
        vector<int> sorted = org;
        sort(sorted.begin(), sorted.end());  // 离散化用

        for(int i = 0; i < n; ++i){
            // 获取 org[i] 在排序后数组中的离散化位置（从1开始）
            int position = lower_bound(sorted.begin(), sorted.end(), org[i]) - sorted.begin() + 1;
            add(position, true);  // 更新递增子序列信息
            // 将值域反转：大值变小，小值变大，从而用相同结构处理递减子序列
            add((int)sorted.size() - position + 1, false);
        }
    }

    // 获取最长子序列长度：pattern == true 为 LIS，false 为 LDS
    int getLength(bool pattern){
        return getMax((int)increase.size() - 1, pattern);
    }
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string line;
    getline(cin, line);
    stringstream ss(line);
    vector<int> nums;
    int x;
    while(ss >> x) nums.push_back(x);

    Analyzer analyzer(nums);
    cout << analyzer.getLength(false) << endl << analyzer.getLength(true) << endl;
    return 0;
}
