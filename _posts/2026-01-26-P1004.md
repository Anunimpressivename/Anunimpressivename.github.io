# P1004
# 最终代码
```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n; cin >> n;
    vector<vector<int>> value(n, vector<int>(n, 0));
    for(int x, y, v; cin >> x >> y >> v, x; )
        value[x-1][y-1] = v;

    vector<vector<vector<int>>> dp(2*n, vector<vector<int>>(n, vector<int>(n)));
    dp[0][0][0] = value[0][0];

    for(int k = 1; k <= 2*n-2; ++k){
        for(int i1 = 0; i1 < n; ++i1){
            for(int i2 = 0; i2 < n; ++i2) {
                int j1 = k - i1, j2 = k - i2;
                if(j1 < 0 || j1 >= n || j2 < 0 || j2 >= n) continue;
                int val = (i1 == i2)? value[i1][j1]: value[i1][j1] + value[i2][j2];
                int& cur = dp[k][i1][i2];
                if(i1) cur = max(cur, dp[k-1][i1-1][i2]);
                if(i2) cur = max(cur, dp[k-1][i1][i2-1]);
                if(i1 && i2) cur = max(cur, dp[k-1][i1-1][i2-1]);
                cur = max(cur, dp[k-1][i1][i2]) + val;
            }
        }
    }

    cout << dp[2*n-2][n-1][n-1] << endl;
    return 0;
}
```
